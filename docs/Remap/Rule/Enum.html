def self.call(&block)
  unless block
    raise ArgumentError, "no block given"
  end

  new.tap { _1.execute(&block) }
end

def from(*keys, to:)
  value = Some(to)

  keys.each do |key|
    mappings[key] = value
    mappings[to] = value
  end
end

def get(key, &error)
  unless error
    return get(key) { raise Error, _1 }
  end

  self[key].bind { return _1 }.or do
    error["Enum key [#{key}] not found among [#{mappings.keys.inspect}]"]
  end
end
