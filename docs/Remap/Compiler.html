def self.call(&block)
  unless block
    return Rule::Void.new
  end

  new.tap { _1.instance_exec(&block) }.rule
end

def map(*path, to: EMPTY_ARRAY, backtrace: Kernel.caller, &block)
  add Rule::Map::Strict.call(
    path: {
      output: [to].flatten,
      input: path.flatten
    },
    backtrace: backtrace,
    rule: call(&block))
end

def wrap(type, &block)
  unless block
    raise ArgumentError, "#wrap requires a block"
  end

  add Rule::Wrap.new(type: type, rule: call(&block))
rescue Dry::Struct::Error => e
  raise ArgumentError, e.message
end
